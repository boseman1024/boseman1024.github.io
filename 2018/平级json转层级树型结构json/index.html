层级关系数据转树型结构JSON

Written by 轴心 with ♥ on May 1, 2018 in 睁眼瞎

[{
    "id": 0,
    "name": "home",

    "component": "Gate",
    "iconcls": "el-icon-menu"
}, {
    "id": 1,
    "name": "user",
    "url": "0",
    "path": "",
    "iconcls": "el-icon-document"
}, {
    "id": 2,
    "name": "User",
    "url": "1",
    "path": "/user/user",
    "component": "user/User",
    "parentid": 1
}, {
    "id": 3,
    "name": "Role",
    "url": "1",
    "path": "/user/role",
    "component": "user/Role",
    "parentid": 1
}, {
    "id": 7,
    "name": "Permission",
    "url": "1",
    "path": "/user/permission",
    "component": "user/Permission",
    "parentid": 1
}, {
    "id": 8,
    "name": "Menu",
    "url": "1",
    "path": "/user/menu",
    "component": "user/Menu",
    "parentid": 1
}]

通过递归的方式转换：

const getJsonTree = function(data,parentid){
    let tree=[]
    for(let i=0;i<data.length;i++){ 
        let node=data[i]
        if(node.parentid==parentid ){ 
            let newNode={
                id:node.id,
                name:node.name,
                path:node.path, 
                component:node.component,
                meta:{
                    url:node.url,
                    iconcls:node.iconcls,
                    parentid:node.parentid
                },
                children:getJsonTree(data,node.id)
            }
            tree.push(newNode)             
        }
    }
    return tree
}
