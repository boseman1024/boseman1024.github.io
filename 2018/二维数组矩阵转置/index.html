普通转置法

int translate(EleArray *eleArray,EleArray *result){
    result->rowNum = eleArray->colNum;
    result->colNum = eleArray->rowNum;
    result->notZeroNum = eleArray->notZeroNum;
    int n;
    if(eleArray->notZeroNum){
        int index=0;//转置后数组个数
        int col,notZeroIndex;//列数，非零序列 
        for(col=0;col<=eleArray->colNum;col++){
            for(notZeroIndex=0;notZeroIndex<eleArray->notZeroNum;notZeroIndex++){
                if(eleArray->data[notZeroIndex].col==col){
                    result->data[index].row = eleArray->data[notZeroIndex].col;
                    result->data[index].col = eleArray->data[notZeroIndex].row;
                    result->data[index].value = eleArray->data[notZeroIndex].value;
                    index++;
                }
            }
        }
    }
    return 1;
}

快速转置法

int fastTranslate(EleArray *eleArray,EleArray *result){
    printf("===========快速转置=========\n");
    result->rowNum = eleArray->colNum;
    result->colNum = eleArray->rowNum;
    result->notZeroNum = eleArray->notZeroNum;
    if(eleArray->notZeroNum){
        int num[eleArray->colNum],cpot[eleArray->colNum],col;
        for(col=0;col<eleArray->colNum;col++){
            num[col]=0;
        }
        for(col=0;col<eleArray->notZeroNum;col++){
            num[eleArray->data[col].col]++;
        }
        cpot[0]=0;
        for(col=1;col<eleArray->colNum;col++){
            cpot[col] = cpot[col-1]+num[col-1];
        }
        int index=0,notZeroIndex;
        for(notZeroIndex=0;notZeroIndex<eleArray->notZeroNum;notZeroIndex++){
            col = eleArray->data[notZeroIndex].col;
            index = cpot[col];
            result->data[index].row = eleArray->data[notZeroIndex].col;
            result->data[index].col = eleArray->data[notZeroIndex].row;
            result->data[index].value = eleArray->data[notZeroIndex].value;
            cpot[col]++;
            index++;
        }
    }
    return 1;
}
