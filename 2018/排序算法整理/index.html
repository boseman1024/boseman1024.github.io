冒泡排序

public static void Bubble(int num[]) {
    for(int i=0;i<num.length-1;i++) {
        for(int j=0;j<num.length-i-1;j++) {
            if(num[j]>num[j+1]) {
                int temp = num[j+1];
                num[j+1] = num[j];
                num[j] = temp;
            }
        }
    }
}

选择排序

public static void Selection(int num[]) {
    for(int i=0;i<num.length;i++) {
        //最小元素的下标
        int min = i;
        //找出最小元素
        for(int j=i;j<num.length;j++) {
            if(num[j]<num[min]) {
                min = j;
            }
        }
        //交换元素
        int temp = num[i];
        num[i] = num[min];
        num[min] = temp;
    }
    show(num);
}

直接插入排序

public static void StraightInsert(int num[]) {
    for(int i=1;i<num.length;i++) {
        int temp = num[i];
        int j=i;
        while(j>0&&temp<num[j-1]) {
            num[j] = num[j-1];
            j--;
        }
        num[j]=temp;
    }
}

二分插入排序

public static void DichotomyInsert(int num[]) {
    for(int i=1;i<num.length;i++) {
        int low = 0;
        int high = i-1;
        int temp = num[i];
        while(low<=high) {
            int mid = (high+low)/2; //取中间点
            if(temp>num[mid]) {
                low = mid+1; //插入点在右半区间
            }else {
                high = mid-1; //插入点在左半区间
            }
        }
        //插入点在low，low到i为止的所有元素向右移动一位
        for(int j=i-1;j>=low;j--) {
            num[j+1] = num[j];
        }
        //插入low位置
        num[low] = temp;
    }
}
